358  private static String computeSignature(String message, String key) throws GeneralSecurityException, UnsupportedEncodingException {
359  // Get an hmac_sha1 key from the raw key bytes
360  byte[] keyBytes = key.getBytes();
361  SecretKeySpec signingKey = new SecretKeySpec(keyBytes, "HmacSHA1");
362 
363  // Get an hmac_sha1 Mac instance and initialize with the signing key
364  Mac mac = Mac.getInstance("HmacSHA1");
365  mac.init(signingKey);
366 
367  // Compute the hmac on input data bytes
368  byte[] rawHmac = mac.doFinal(message.getBytes());
369 
370  StringBuilder string = new StringBuilder();
371  for (byte b : rawHmac) {
372  String hexString = Integer.toHexString(0x00FF & b);
373  string.append(hexString.length() == 1 ? "0" + hexString : hexString);
374  }
375 
376  return string.toString();
